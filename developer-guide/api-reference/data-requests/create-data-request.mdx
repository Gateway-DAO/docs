You can create a data request using the `createDataRequest` mutation. You will need a Data Request Template ID.

You can choose from the available templates in the Gateway Protocol dashboard or create a new one.

```graphql
mutation {
  createDataRequest(
    input: {
      dataRequestTemplateId: "YOUR_DATA_REQUEST_TEMPLATE_ID"
      owner: { type: GATEWAY_ID, value: "OWNER_GATEWAY_ID" }
      dataUse: "YOUR_DATA_USE_PURPOSE"
    }
  ) {
    arweaveUrl
    id
    status
    dataUse
    verifierOrganization {
      id
      gatewayId
    }
  }
}
```

<RequestExample>

```ts Node.js
const axios = require("axios");

let data = JSON.stringify({
  query: `mutation {
    createDataRequest(input: {
      dataRequestTemplateId: "YOUR_DATA_REQUEST_TEMPLATE_ID",
      owner: {
        type: GATEWAY_ID,
        value: "OWNER_GATEWAY_ID"
      },
      dataUse: "YOUR_DATA_USE_PURPOSE"
    }) {
      arweaveUrl
      id
      status
      dataUse
      verifierOrganization {
        id
        gatewayId
      }
    }
  }`,
  variables: {},
});

let config = {
  method: "post",
  maxBodyLength: Infinity,
  url: "https://protocol.mygateway.xyz/graphql",
  headers: {
    "Content-Type": "application/json",
    "x-api-key": "YOUR_API_KEY",
    Authorization: "Bearer YOUR_AUTHENTICATION_TOKEN",
  },
  data: data,
};

axios
  .request(config)
  .then((response) => {
    console.log(JSON.stringify(response.data));
  })
  .catch((error) => {
    console.log(error);
  });
```
