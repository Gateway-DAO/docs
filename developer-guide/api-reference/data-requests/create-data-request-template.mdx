You can create a data request template using the `createDataRequestTemplate` mutation.

```graphql
mutation {
  createDataRequestTemplate(
    input: {
      title: "Your Data Request Template Title"
      description: "Your Data Request Template Description"
      organization: { type: GATEWAY_ID, value: "Your Organization ID" }
      dataModels: [
        {
          id: "Data Model ID 1"
          required: true
          claimValidations: {
            type: "object"
            properties: { field1: { type: "integer", minimum: 10 } }
            required: ["field1"]
          }
        }
        {
          id: "Data Model ID 2"
          required: true
          claimValidations: {
            type: "object"
            properties: { field2: { type: "integer", minimum: 100 } }
            required: ["field2"]
          }
        }
        {
          id: "Data Model ID 3"
          required: true
          claimValidations: {
            type: "object"
            properties: { field3: { type: "integer", minimum: 5 } }
            required: ["field3"]
          }
        }
      ]
    }
  ) {
    id
    arweaveUrl
    tags
  }
}
```
