You can fetch your transactions using the `myFinancialTransactions` query. You can specify the `skip` and `take` parameters to skip and take a certain number of transactions. You can also specify the `organizationId` to filter the transactions based on the organization.

```graphql
query my_transactions($skip: Int, $take: Int, $organizationId: String) {
  myFinancialTransactions(
    skip: $skip
    take: $take
    organizationId: $organizationId
  ) {
    id
    value
    type
    transactionId
    createdAt
    action
    memo
  }
}
```

<RequestExample>

```ts Node.js
const axios = require("axios");
let data = JSON.stringify({
  query: `query my_transactions($skip: Int, $take: Int, $organizationId: String) {
  myFinancialTransactions(
    skip: $skip
    take: $take
    organizationId: $organizationId
  ) {
    id
    value
    type
    transactionId
    createdAt
    action
    memo
  }
}`,
  variables: { skip: 0, take: 1000 },
});

let config = {
  method: "post",
  maxBodyLength: Infinity,
  url: "https://protocol.mygateway.xyz/graphql",
  headers: {
    "X-Api-Key": "<YOUR_API_KEY>",
    Authorization: "Bearer <YOUR_AUTHENTICATION_TOKEN>",
    "Content-Type": "application/json",
  },
  data: data,
};

axios
  .request(config)
  .then((response) => {
    console.log(JSON.stringify(response.data));
  })
  .catch((error) => {
    console.log(error);
  });
```

</RequestExample>
