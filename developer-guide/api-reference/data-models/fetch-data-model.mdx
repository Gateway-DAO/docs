You can fetch a data model by its ID.

```graphql
query {
  dataModel(id: "9f27397e-27f2-4c30-b1b7-829371de4df5") {
    id
    title
    description
    arweaveUrl
    createdBy {
      id
      gatewayId
      profilePicture
      createdAt
      displayName
    }
    organization {
      id
      gatewayId
      name
      image
    }
    tags
    consumptionPrice
    pdasIssuedCount
    uniqueIssuersCount
    revenueGenerated
    schema
    createdAt
    permissioning
    allowedUsers {
      id
      gatewayId
      profilePicture
      createdAt
      displayName
    }
    allowedOrganizations {
      id
      gatewayId
      name
      image
    }
  }
}
```

```ts Node.js
let data = JSON.stringify({
  query: `query {
  dataModel(id: "9f27397e-27f2-4c30-b1b7-829371de4df5") {
    id
    title
    description
    arweaveUrl
    createdBy {
        id
        gatewayId
        profilePicture
        createdAt
        displayName
    }
    organization {
        id
        gatewayId
        name
        image
    }
    tags
    consumptionPrice
    pdasIssuedCount
    uniqueIssuersCount
    revenueGenerated
    schema
    createdAt
    permissioning
    allowedUsers {
        id
        gatewayId
        profilePicture
        createdAt
        displayName
    }
    allowedOrganizations {
        id
        gatewayId
        name
        image
    }
  }
}`,
  variables: {},
});

let config = {
  method: "post",
  maxBodyLength: Infinity,
  url: "https://protocol.mygateway.xyz/graphql",
  headers: {
    "Content-Type": "application/json",
    "x-api-key": "m9Y5ntNcTlwQ2LbRpYr6K_VhxJXuZJ6Q",
  },
  data: data,
};

axios
  .request(config)
  .then((response) => {
    console.log(JSON.stringify(response.data));
  })
  .catch((error) => {
    console.log(error);
  });
```
