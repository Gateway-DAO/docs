You can fetch all organizations that the user is a part of. This will return the organizations that the user is a part of.

```graphql
query {
  organizations {
    id
    gatewayId
    accesses {
      id
      role
      user {
        id
        gatewayId
      }
    }
    verified
  }
}
```

<RequestExample>

```ts Node.js
const axios = require("axios");
let data = JSON.stringify({
  query: `query {
  organizations {
    id
    gatewayId
    accesses {
        id
        role
        user {
            id
            gatewayId
        }
    }
    verified
  }
}`,
  variables: {},
});

let config = {
  method: "post",
  maxBodyLength: Infinity,
  url: "https://protocol.mygateway.xyz/graphql",
  headers: {
    "Content-Type": "application/json",
    "x-api-key": "m9Y5ntNcTlwQ2LbRpYr6K_VhxJXuZJ6Q",
    Authorization: "Bearer {{ _.token }}",
  },
  data: data,
};

axios
  .request(config)
  .then((response) => {
    console.log(JSON.stringify(response.data));
  })
  .catch((error) => {
    console.log(error);
  });
```
